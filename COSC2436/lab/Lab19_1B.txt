package lab19;

 * This program will create and process an ArrayList of Strings
 */

import java.io.File;
import java.io.FileNotFoundException;
import java.util.ArrayList;
import java.util.Scanner;


public class Lab19_1B {

	public static void main(String[] args) throws FileNotFoundException {
		File file = new File("Input19_1B.txt");
		Scanner inFile = new Scanner(file);
		
		ArrayList<String> arr = new ArrayList<>();
		// Read an array of strings from file
		while (inFile.hasNext()) {
			String s;
			s = inFile.next();
			arr.add(s);
		}
		System.out.print("ArrayList: ");
		printArray(arr);
		System.out.println();
		
		System.out.print("Strings with a length larger than 5: ");
		int count = countStrings(arr); // call count strings with a length larger than 5
		System.out.println();
		
		System.out.println("Count: " + count);
	}

	public static void printArray(ArrayList<String> arr) {
		for (String temp : arr) {
			System.out.print(temp + " ");
		}
		
	}
	
	public static int countStrings(ArrayList<String> arr) {
		int count = 0;
		for (String temp : arr) {
			if (temp.length() > 5) {
				count++;
				System.out.print(temp + " ");
			}
		}
		return count;
	}
}
