import java.io.File;
import java.io.FileNotFoundException;
import java.util.ArrayList;
import java.util.Scanner;


public class Lab19_1C {

	public static void main(String[] args) throws FileNotFoundException {
		File file = new File("Input19_1C.txt");
		Scanner inFile = new Scanner(file);
		
		ArrayList<Basket> arr = new ArrayList<>();
		// Read baskets from file
		while (inFile.hasNext()) {
			String tempType = inFile.next();
			int tempNum = inFile.nextInt();
			arr.add(new Basket(tempType, tempNum));
		}
	
		// Calculate average
		double average;
		double sum = 0;
		for (Basket b : arr) {
			sum += b.getNumber();
		}
		average = sum / arr.size();
		System.out.println("Average: " + String.format("%.2f",average));
		
		// Print information of all baskets
		System.out.println("Information of baskets:");

		for (Basket b : arr) {
			System.out.println(b);
		}
	}

}


// Basket class
class Basket {
	private String type;
	private int number;
	public Basket(String type, int num) {
		this.type = type;
		this.number = num;
	}
	public int getNumber() {
		return this.number;
	}
	public String getType() {
		return this.type;
	}
	public String toString() {
		return "Basket - Type:" + this.type + " - Number:" + this.number;  
	}
}
