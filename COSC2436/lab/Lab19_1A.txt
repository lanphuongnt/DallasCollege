package lab19;

 * This program will create and process an ArrayList of Strings
 */

import java.io.File;
import java.io.FileNotFoundException;
import java.util.ArrayList;
import java.util.Scanner;


public class Lab19_1A {

	public static void main(String[] args) throws FileNotFoundException {
		File file = new File("Input19_1A.txt");
		Scanner inFile = new Scanner(file);
		
		ArrayList<String> arr = new ArrayList<>();
		// Read an array of strings from file
		while (inFile.hasNext()) {
			String s;
			s = inFile.next();
			arr.add(s);
		}
		
		// Print the array list
		System.out.print("Array of Strings: ");
		for (String temp : arr) {
			System.out.print(temp + " ");
		}
		System.out.println();
		
		// Print all items with a length less than the input number
		printItems(arr);
	}

	public static void printItems(ArrayList<String> arr) {
		int number;
		Scanner scanner = new Scanner(System.in);
		System.out.print("Enter an integer: ");
		number = scanner.nextInt();
		System.out.print("Strings: ");
		for (String temp : arr) {
			if (temp.length() < number) {
				System.out.print(temp + " ");
			}
		}
	}
}
