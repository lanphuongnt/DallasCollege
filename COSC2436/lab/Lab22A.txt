package lab22;

import java.util.ArrayList;
import java.util.Random;
import java.util.Scanner;
import java.util.TreeSet;

/**
 * Lab22A
 * @author Phuong Nguyen 
 * This program will create an ArrayList and a TreeSet of 1000 items and will compare the time it takes to fill them and search for a number
 */

public class Lab22A {

	public static void main(String[] args) {
		ArrayList<Integer> arr = new ArrayList<>();
		TreeSet<Integer> set = new TreeSet<>();
		fill(arr, set);
		
		// Print ArrayList
		System.out.print("ArrayList: ");
		for (Integer i : arr) {
			System.out.print(i + " ");
		}
		System.out.println();
		// Print TreeSet
		System.out.print("TreeSet: ");
		for (Integer i : set) {
			System.out.print(i + " ");
		}
		System.out.println();
		
		
		search(arr, set);
	}

	public static void fill(ArrayList<Integer> arr, TreeSet<Integer> set) {
		Random rand = new Random();
		long startTime, endTime;
		// Fill the ArrayList with 1000 random values between 0 and 2000
		startTime = System.nanoTime(); // start time
		for (int i = 0; i < 1000; i++) {
			arr.add(rand.nextInt(2001));
		}
		endTime = System.nanoTime();
		System.out.println("Execution time for filling ArrayList is " + String.format("%,d",endTime - startTime) + " nano seconds");

		// Fill the TreeSet with 1000 random values between 0 and 2000
		startTime = System.nanoTime();
		for (int i = 0; i < 1000; i++) {
			set.add(rand.nextInt(2001));
		}
		endTime = System.nanoTime();
		System.out.println("Execution time for filling TreeSet is " + String.format("%,d",endTime - startTime) + " nano seconds");

		// Fill the TreeSet until its size is 1000
		while (set.size() < 1000) {
			set.add(rand.nextInt(2001));
		}
	}

	public static void search(ArrayList<Integer> arr, TreeSet<Integer> set) {
		Scanner scan = new Scanner(System.in);
		long startTime, endTime;
		// Get number from user
		System.out.print("Enter an integer between 1 and 2000: ");
		int number = scan.nextInt();

		startTime = System.nanoTime();
		// Use contains method to search in ArrayList
		if (arr.contains(number)) {
			System.out.println("The ArrayList contains " + number);
		} else {
			System.out.println("The ArrayList does not contain " + number);
		}
		endTime = System.nanoTime();
		System.out.println("Execution time for searching number in ArrayList is " + String.format("%,d",endTime - startTime) + " nano seconds");

		// Use contains method to search in TreeSet
		if (set.contains(number)) {
			System.out.println("The TreeSet contains " + number);
		} else {
			System.out.println("The TreeSet does not contain " + number);
		}
		endTime = System.nanoTime();
		System.out.println("Execution time for searching number in TreeSet is " + String.format("%,d",endTime - startTime) + " nano seconds");
	}
}
